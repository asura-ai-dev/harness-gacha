# Task 1004: manifest.json バリデーター

## 概要

manifest.json の必須フィールドを検証するバリデーターを `src/data/manifest.rs` に実装する。F-8（manifest バリデーション）に対応。

## 詳細

### 1. `src/data/manifest.rs` の実装

```rust
use std::path::Path;
use crate::models::manifest::Manifest;

/// manifest.json のバリデーション結果
#[derive(Debug)]
pub struct ValidationResult {
    pub valid: bool,
    pub errors: Vec<String>,
}

/// manifest.json ファイルを読み込みパースする
pub fn load_manifest(path: &Path) -> Result<Manifest, String> {
    let content = std::fs::read_to_string(path)
        .map_err(|e| format!("ファイル読み込みエラー: {}", e))?;
    serde_json::from_str::<Manifest>(&content)
        .map_err(|e| format!("JSON パースエラー: {}", e))
}

/// Manifest の必須フィールドと整合性を検証する
pub fn validate_manifest(manifest: &Manifest) -> ValidationResult {
    let mut errors = Vec::new();

    // id: 英小文字、数字、ハイフンのみ
    if !manifest.id.chars().all(|c| c.is_ascii_lowercase() || c.is_ascii_digit() || c == '-') {
        errors.push(format!("id '{}' に不正な文字が含まれています（英小文字、数字、- のみ許可）", manifest.id));
    }
    if manifest.id.is_empty() {
        errors.push("id が空です".to_string());
    }

    // name: 空でないこと
    if manifest.name.trim().is_empty() {
        errors.push("name が空です".to_string());
    }

    // version: 空でないこと
    if manifest.version.trim().is_empty() {
        errors.push("version が空です".to_string());
    }

    // summary: 空でないこと
    if manifest.summary.trim().is_empty() {
        errors.push("summary が空です".to_string());
    }

    // author.name: 空でないこと
    if manifest.author.name.trim().is_empty() {
        errors.push("author.name が空です".to_string());
    }

    // targets: 1 件以上
    if manifest.targets.is_empty() {
        errors.push("targets が空です（最低 1 件必要）".to_string());
    }

    // contents: 1 件以上
    if manifest.contents.is_empty() {
        errors.push("contents が空です（最低 1 件必要）".to_string());
    }

    // install.method: "copy" | "script" | "manual" のいずれか
    let valid_methods = ["copy", "script", "manual"];
    if !valid_methods.contains(&manifest.install.method.as_str()) {
        errors.push(format!(
            "install.method '{}' は不正です（copy, script, manual のいずれか）",
            manifest.install.method
        ));
    }

    // license.type: 空でないこと
    if manifest.license.license_type.trim().is_empty() {
        errors.push("license.type が空です".to_string());
    }

    ValidationResult {
        valid: errors.is_empty(),
        errors,
    }
}

/// contents に列挙されたファイルの存在を検証する（ローカル pack 用）
pub fn validate_contents(manifest: &Manifest, pack_root: &Path) -> Vec<String> {
    let mut missing = Vec::new();
    for entry in &manifest.contents {
        let file_path = pack_root.join(entry);
        if !file_path.exists() {
            missing.push(format!("contents に記載された '{}' が存在しません", entry));
        }
    }
    missing
}
```

### 2. テストの追加

`src/data/manifest.rs` の末尾に `#[cfg(test)] mod tests` を追加する。

```rust
#[cfg(test)]
mod tests {
    use super::*;
    use crate::models::catalog_entry::{Author, Target, InstallInfo, LicenseInfo};
    use crate::models::permissions::Permissions;

    fn valid_manifest() -> Manifest {
        Manifest {
            schema_version: "1.0".to_string(),
            id: "test-pack".to_string(),
            name: "Test Pack".to_string(),
            version: "1.0.0".to_string(),
            summary: "A test pack".to_string(),
            description: None,
            author: Author { name: "Test".to_string(), url: None, email: None },
            targets: vec![Target { tool: "claude-code".to_string(), version_range: ">=1.0.0".to_string() }],
            contents: vec!["skills/test.md".to_string()],
            permissions: Permissions { shell: false, network: false, filesystem_read: true, filesystem_write: false, git: false },
            install: InstallInfo { method: "copy".to_string(), entrypoint: None, steps: None },
            license: LicenseInfo { license_type: "commercial".to_string(), text_url: None, spdx: None },
            tags: None,
            risks: None,
            post_install: None,
            homepage: None,
            repository: None,
        }
    }

    #[test]
    fn test_valid_manifest() {
        let result = validate_manifest(&valid_manifest());
        assert!(result.valid);
        assert!(result.errors.is_empty());
    }

    #[test]
    fn test_empty_id() {
        let mut m = valid_manifest();
        m.id = "".to_string();
        let result = validate_manifest(&m);
        assert!(!result.valid);
        assert!(result.errors.iter().any(|e| e.contains("id")));
    }

    #[test]
    fn test_invalid_id_chars() {
        let mut m = valid_manifest();
        m.id = "Test Pack".to_string();
        let result = validate_manifest(&m);
        assert!(!result.valid);
    }

    #[test]
    fn test_empty_targets() {
        let mut m = valid_manifest();
        m.targets = vec![];
        let result = validate_manifest(&m);
        assert!(!result.valid);
        assert!(result.errors.iter().any(|e| e.contains("targets")));
    }

    #[test]
    fn test_invalid_install_method() {
        let mut m = valid_manifest();
        m.install.method = "invalid".to_string();
        let result = validate_manifest(&m);
        assert!(!result.valid);
        assert!(result.errors.iter().any(|e| e.contains("install.method")));
    }
}
```
