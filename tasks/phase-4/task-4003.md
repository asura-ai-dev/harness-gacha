### 1. `src/ui/library.rs`

レイアウト分割（垂直 3 分割）:

1. ヘッダー: 3 行固定 - "My Library"
2. コンテンツ: 残り全部 - entitlement 一覧リスト
3. フッター: 3 行固定 - キーバインド

```rust
use ratatui::prelude::*;
use ratatui::widgets::{Block, Borders, List, ListItem, ListState, Paragraph};
use crate::app::App;
use crate::ui::theme::Theme;
use crate::data::catalog::find_pack_by_id;
use crate::data::entitlement;

pub fn render(frame: &mut Frame, app: &App, theme: &Theme) {
    let chunks = Layout::default()
        .direction(Direction::Vertical)
        .constraints([
            Constraint::Length(3),
            Constraint::Fill(1),
            Constraint::Length(3),
        ])
        .split(frame.area());

    // ヘッダー
    let header = Paragraph::new("My Library")
        .style(Style::default().fg(theme.accent).add_modifier(Modifier::BOLD))
        .block(Block::default().borders(Borders::ALL).border_style(theme.border_style()).style(theme.panel_style()));
    frame.render_widget(header, chunks[0]);

    // コンテンツ: entitlement 一覧
    render_list(frame, chunks[1], app, theme);

    // フッター
    let footer = Paragraph::new("[Enter] Install 詳細  [b] 戻る")
        .style(theme.footer_text_style())
        .block(Block::default().borders(Borders::ALL).border_style(theme.border_style()));
    frame.render_widget(footer, chunks[2]);
}
```

#### リスト描画

entitlements から active なもののみを取り出し、catalog から pack 詳細を引いて表示する。

```rust
fn render_list(frame: &mut Frame, area: Rect, app: &App, theme: &Theme) {
    // entitlements を取得（Option の場合は unwrap or default）
    let store = match &app.entitlements {
        Some(s) => s,
        None => {
            let msg = Paragraph::new("購入済み pack はありません")
                .style(theme.text_style())
                .alignment(Alignment::Center);
            frame.render_widget(msg, area);
            return;
        }
    };

    let active = entitlement::active_entitlements(store);
    if active.is_empty() {
        let msg = Paragraph::new("購入済み pack はありません")
            .style(theme.text_style())
            .alignment(Alignment::Center);
        frame.render_widget(msg, area);
        return;
    }

    let items: Vec<ListItem> = active.iter()
        .map(|ent| {
            let pack = find_pack_by_id(&app.catalog, &ent.pack_id);
            let name = pack.map(|p| p.name.as_str()).unwrap_or(&ent.pack_id);
            let version = pack.map(|p| p.version.as_str()).unwrap_or("?");

            let install_status = if ent.installed { "Installed" } else { "Not Installed" };
            let install_color = if ent.installed { theme.owned } else { theme.text_secondary };

            // update 有無: installed_version と catalog の version を比較
            let update_status = if ent.installed {
                match (&ent.installed_version, pack) {
                    (Some(iv), Some(p)) if iv != &p.version => "Update: available",
                    _ => "Update: none",
                }
            } else {
                ""
            };
            let update_color = if update_status == "Update: available" { theme.warning } else { theme.text_secondary };

            let line = Line::from(vec![
                Span::styled(format!("  {:<25}", name), Style::default().fg(theme.text_primary)),
                Span::styled(format!("v{:<10}", version), Style::default().fg(theme.text_secondary)),
                Span::styled(format!("{:<16}", install_status), Style::default().fg(install_color)),
                Span::styled(update_status.to_string(), Style::default().fg(update_color)),
            ]);
            ListItem::new(line)
        })
        .collect();

    let list = List::new(items)
        .block(Block::default().borders(Borders::LEFT | Borders::RIGHT).border_style(theme.border_style()))
        .highlight_style(theme.highlight_style())
        .highlight_symbol("> ");

    let mut state = ListState::default();
    state.select(Some(app.library_state.selected_index));
    frame.render_stateful_widget(list, area, &mut state);
}
```

### 3. `src/app.rs` の更新

`handle_screen_action` に `Screen::Library` のハンドラを追加する。

```rust
Screen::Library => {
    match action {
        Action::Up => {
            if self.library_state.selected_index > 0 {
                self.library_state.selected_index -= 1;
            }
        }
        Action::Down => {
            let max = self.active_entitlement_count().saturating_sub(1);
            if self.library_state.selected_index < max {
                self.library_state.selected_index += 1;
            }
        }
        Action::Enter => {
            // 選択中の entitlement の pack_id を selected_pack_id に設定し、InstallDetail に遷移
            if let Some(store) = &self.entitlements {
                let active: Vec<&_> = crate::data::entitlement::active_entitlements(store);
                if let Some(ent) = active.get(self.library_state.selected_index) {
                    self.selected_pack_id = Some(ent.pack_id.clone());
                    self.navigate_to(Screen::InstallDetail);
                }
            }
        }
        _ => {}
    }
}
```

/// active な entitlement の件数を返す
fn active_entitlement_count(&self) -> usize {
    match &self.entitlements {
        Some(store) => crate::data::entitlement::active_entitlements(store).len(),
        None => 0,
    }
}
