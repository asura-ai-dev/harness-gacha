### 2. `src/data/accounting.rs`

```rust
use std::path::Path;
use std::collections::HashMap;
use crate::models::accounting::{AccountingData, Transaction, CreatorShare};

/// accounting.json を読み込む
/// ファイルが存在しない場合はデフォルト値（空）を返す
pub fn load_accounting(path: &Path) -> AccountingData {
    match std::fs::read_to_string(path) {
        Ok(content) => {
            serde_json::from_str::<AccountingData>(&content).unwrap_or_else(|e| {
                eprintln!("accounting.json パースエラー: {}", e);
                default_accounting()
            })
        }
        Err(_) => default_accounting(),
    }
}

/// accounting.json に書き込む
pub fn save_accounting(path: &Path, data: &AccountingData) -> Result<(), String> {
    let json = serde_json::to_string_pretty(data)
        .map_err(|e| format!("JSON シリアライズエラー: {}", e))?;
    std::fs::write(path, json)
        .map_err(|e| format!("ファイル書き込みエラー: {}", e))
}

fn default_accounting() -> AccountingData {
    AccountingData {
        transactions: Vec::new(),
        creator_shares: HashMap::new(),
    }
}

/// 指定 pack_id の gross sales（purchase の合計額）を算出する
pub fn gross_sales_for_pack(data: &AccountingData, pack_id: &str) -> i64 {
    data.transactions.iter()
        .filter(|t| t.pack_id == pack_id && t.tx_type == "purchase")
        .map(|t| t.amount)
        .sum()
}

/// 指定 pack_id の refund 合計額を算出する
pub fn refunds_for_pack(data: &AccountingData, pack_id: &str) -> i64 {
    data.transactions.iter()
        .filter(|t| t.pack_id == pack_id && t.tx_type == "refund")
        .map(|t| t.amount)
        .sum()
}

/// 指定 pack_id の net sales（gross - refund）を算出する
pub fn net_sales_for_pack(data: &AccountingData, pack_id: &str) -> i64 {
    gross_sales_for_pack(data, pack_id) - refunds_for_pack(data, pack_id)
}

/// 指定年月（"YYYY-MM" 形式）の transaction をフィルタして返す
pub fn transactions_for_month<'a>(data: &'a AccountingData, year_month: &str) -> Vec<&'a Transaction> {
    data.transactions.iter()
        .filter(|t| t.timestamp.starts_with(year_month))
        .collect()
}

/// 月次クリエイター分配額を算出する
/// 戻り値: HashMap<creator_name, payout_amount>
/// ロジック:
///   1. 対象月の transaction をフィルタ
///   2. 各 creator の担当 pack ごとに net_sales を算出
///   3. net_sales * share_rate を分配額とする
pub fn calculate_monthly_payouts(
    data: &AccountingData,
    year_month: &str,
) -> HashMap<String, i64> {
    let month_txns = transactions_for_month(data, year_month);
    let mut payouts: HashMap<String, i64> = HashMap::new();

    for (creator_name, share) in &data.creator_shares {
        let mut creator_total: i64 = 0;
        for pack_id in &share.packs {
            // 対象月の該当 pack の net sales
            let gross: i64 = month_txns.iter()
                .filter(|t| t.pack_id == *pack_id && t.tx_type == "purchase")
                .map(|t| t.amount)
                .sum();
            let refund: i64 = month_txns.iter()
                .filter(|t| t.pack_id == *pack_id && t.tx_type == "refund")
                .map(|t| t.amount)
                .sum();
            let net = gross - refund;
            creator_total += (net as f64 * share.share_rate) as i64;
        }
        payouts.insert(creator_name.clone(), creator_total);
    }

    payouts
}
```

### 3. テストの追加

```rust
#[cfg(test)]
mod tests {
    use super::*;
    use crate::models::accounting::{AccountingData, Transaction, CreatorShare};

    fn sample_data() -> AccountingData {
        let mut creator_shares = HashMap::new();
        creator_shares.insert("Creator A".to_string(), CreatorShare {
            share_rate: 0.7,
            packs: vec!["pack-a".to_string()],
        });
        creator_shares.insert("Creator B".to_string(), CreatorShare {
            share_rate: 0.6,
            packs: vec!["pack-b".to_string()],
        });

        AccountingData {
            transactions: vec![
                Transaction {
                    pack_id: "pack-a".to_string(),
                    user_id: "user-1".to_string(),
                    amount: 1000,
                    tx_type: "purchase".to_string(),
                    timestamp: "2026-04-05T10:00:00Z".to_string(),
                    stripe_payment_id: None,
                },
                Transaction {
                    pack_id: "pack-a".to_string(),
                    user_id: "user-2".to_string(),
                    amount: 1000,
                    tx_type: "purchase".to_string(),
                    timestamp: "2026-04-10T10:00:00Z".to_string(),
                    stripe_payment_id: None,
                },
                Transaction {
                    pack_id: "pack-a".to_string(),
                    user_id: "user-1".to_string(),
                    amount: 1000,
                    tx_type: "refund".to_string(),
                    timestamp: "2026-04-12T10:00:00Z".to_string(),
                    stripe_payment_id: None,
                },
                Transaction {
                    pack_id: "pack-b".to_string(),
                    user_id: "user-3".to_string(),
                    amount: 2000,
                    tx_type: "purchase".to_string(),
                    timestamp: "2026-04-15T10:00:00Z".to_string(),
                    stripe_payment_id: None,
                },
            ],
            creator_shares,
        }
    }

    #[test]
    fn test_gross_sales() {
        let data = sample_data();
        assert_eq!(gross_sales_for_pack(&data, "pack-a"), 2000);
        assert_eq!(gross_sales_for_pack(&data, "pack-b"), 2000);
        assert_eq!(gross_sales_for_pack(&data, "nonexistent"), 0);
    }

    #[test]
    fn test_refunds() {
        let data = sample_data();
        assert_eq!(refunds_for_pack(&data, "pack-a"), 1000);
        assert_eq!(refunds_for_pack(&data, "pack-b"), 0);
    }

    #[test]
    fn test_net_sales() {
        let data = sample_data();
        assert_eq!(net_sales_for_pack(&data, "pack-a"), 1000);
        assert_eq!(net_sales_for_pack(&data, "pack-b"), 2000);
    }

    #[test]
    fn test_monthly_payouts() {
        let data = sample_data();
        let payouts = calculate_monthly_payouts(&data, "2026-04");
        // Creator A: pack-a net = 2000 - 1000 = 1000, payout = 1000 * 0.7 = 700
        assert_eq!(*payouts.get("Creator A").unwrap(), 700);
        // Creator B: pack-b net = 2000 - 0 = 2000, payout = 2000 * 0.6 = 1200
        assert_eq!(*payouts.get("Creator B").unwrap(), 1200);
    }

    #[test]
    fn test_monthly_payouts_empty_month() {
        let data = sample_data();
        let payouts = calculate_monthly_payouts(&data, "2026-05");
        assert_eq!(*payouts.get("Creator A").unwrap(), 0);
        assert_eq!(*payouts.get("Creator B").unwrap(), 0);
    }
}
```
